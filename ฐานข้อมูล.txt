CREATE DATABASE restaurant_db;
USE restaurant_db;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(100) UNIQUE NOT NULL,
  password VARCHAR(255) NOT NULL,
  role ENUM('customer','shop','admin') NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE shops (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  is_open BOOLEAN DEFAULT TRUE,
  open_time TIME,
  close_time TIME
);



CREATE TABLE menus (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  price DECIMAL(10,2) NOT NULL,
  image VARCHAR(255),
  is_recommended BOOLEAN DEFAULT FALSE
);

CREATE TABLE menu_stats (
  menu_id INT PRIMARY KEY,
  order_count INT DEFAULT 0,
  FOREIGN KEY (menu_id) REFERENCES menus(id)
);

CREATE TABLE queue_status (
  id INT PRIMARY KEY,
  remaining_queue INT NOT NULL
);

CREATE TABLE orders (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  total_price DECIMAL(10,2) NOT NULL,
  status ENUM('pending','cooking','done','cancel') DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE order_items (
  id INT AUTO_INCREMENT PRIMARY KEY,
  order_id INT NOT NULL,
  menu_name VARCHAR(100) NOT NULL,
  price DECIMAL(10,2) NOT NULL,
  quantity INT NOT NULL,
  FOREIGN KEY (order_id) REFERENCES orders(id)
);

-- ==========================================
-- สเต็ปที่ 2: อัปเกรดตาราง menus เดิม
-- ==========================================
-- เพิ่มคอลัมน์ category_id (เชื่อมกับตารางหมวดหมู่) และ description (คำอธิบาย)
ALTER TABLE menus
ADD COLUMN category_id INT DEFAULT NULL AFTER id,
ADD COLUMN description TEXT NULL AFTER name,
ADD FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE SET NULL;

-- ==========================================
-- สเต็ปที่ 3: สร้างตารางตัวเลือกเสริม / ท็อปปิ้ง (Menu Options)
-- ==========================================
CREATE TABLE menu_options (
  id INT AUTO_INCREMENT PRIMARY KEY,
  menu_id INT NOT NULL,
  option_group VARCHAR(100) NOT NULL, -- ชื่อกลุ่มตัวเลือก เช่น 'ระดับความเผ็ด', 'ขนาด', 'ท็อปปิ้ง'
  option_name VARCHAR(100) NOT NULL,  -- ชื่อตัวเลือก เช่น 'ปกติ', 'พิเศษ', 'ไข่ดาว'
  extra_price DECIMAL(10,2) DEFAULT 0, -- ราคาที่ต้องบวกเพิ่ม (เช่น ไข่ดาว +10)
  is_multiple BOOLEAN DEFAULT FALSE,  -- เลือกได้หลายข้อหรือไม่ (เช่น ท็อปปิ้ง=TRUE, ขนาด=FALSE)
  FOREIGN KEY (menu_id) REFERENCES menus(id) ON DELETE CASCADE
);
